

public class PromedioAcademico {

    public static double calcularPromedio(double nota1, double nota2, double nota3) {
        return (nota1 + nota2 + nota3) / 3;
    }

    public static void mostrarResultado(double promedio) {
        if (promedio >= 3.0) {
            System.out.printf("El promedio es %.2f. Aprobado.%n", promedio);
        } else {
            System.out.printf("El promedio es %.2f. Reprobado.%n", promedio);
        }
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Ingresa la primera nota: ");
        double nota1 = scanner.nextDouble();
        System.out.print("Ingresa la segunda nota: ");
        double nota2 = scanner.nextDouble();
        System.out.print("Ingresa la tercera nota: ");
        double nota3 = scanner.nextDouble();
        scanner.close();

        if (Double.isNaN(nota1) || Double.isNaN(nota2) || Double.isNaN(nota3)) {
            System.out.println("Por favor ingresa solo números válidos.");
        } else {
            double promedio = calcularPromedio(nota1, nota2, nota3);
            mostrarResultado(promedio);
        }
    }
}
